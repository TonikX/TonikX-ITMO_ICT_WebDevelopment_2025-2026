    Лабораторная работа LP2 — Система управления автогонками

    Автор: Губанов Егор
    Группа: К3340
    Вариант: 6 — Табло победителей автогонок

    Цель работы

    Реализовать веб-сервис для управления автогонками с использованием Django Web Framework.

    Описание проекта

    Система управления автогонками включает: - Регистрацию пользователей и команд - Управление гонщиками и их профилями - Планирование и проведение гонок - Регистрацию участников на гонки - Ведение результатов и комментариев

    Практическое задание 1 — Создание моделей данных

    1.1 Кастомная модель пользователя

    from django.contrib.auth.models import AbstractUser
from django.db import models

class CustomUser(AbstractUser):
    email = models.EmailField(unique=True)
    phone = models.CharField(max_length=15, blank=True)
    birth_date = models.DateField(null=True, blank=True)
    bio = models.TextField(blank=True)
    avatar = models.ImageField(upload_to='avatars/', null=True, blank=True)
    is_organizer = models.BooleanField(default=False)

    USERNAME_FIELD = 'email'
    REQUIRED_FIELDS = ['username']

    def __str__(self):
        return self.email

    1.2 Модель команды

    class Team(models.Model):
    name = models.CharField(max_length=100, unique=True)
    description = models.TextField(blank=True)
    logo = models.ImageField(upload_to='team_logos/', null=True, blank=True)
    founded_year = models.IntegerField()
    country = models.CharField(max_length=50)
    website = models.URLField(blank=True)
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.name

    1.3 Модель гонщика

    class Driver(models.Model):
    EXPERIENCE_CHOICES = [
        ('beginner', 'Начинающий'),
        ('amateur', 'Любитель'),
        ('professional', 'Профессионал'),
    ]

    CLASS_CHOICES = [
        ('formula1', 'Формула 1'),
        ('formula2', 'Формула 2'),
        ('gt', 'GT'),
        ('touring', 'Туринг'),
    ]

    user = models.OneToOneField(CustomUser, on_delete=models.CASCADE)
    team = models.ForeignKey(Team, on_delete=models.CASCADE, related_name='drivers')
    driver_number = models.IntegerField(unique=True)
    experience = models.CharField(max_length=20, choices=EXPERIENCE_CHOICES)
    driver_class = models.CharField(max_length=20, choices=CLASS_CHOICES)
    car_description = models.TextField()
    bio = models.TextField()
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return f"{self.user.first_name} {self.user.last_name} (#{self.driver_number})"

    1.4 Модель гонки

    class Race(models.Model):
    RACE_TYPE_CHOICES = [
        ('qualifying', 'Квалификация'),
        ('sprint', 'Спринт'),
        ('main', 'Основная гонка'),
        ('endurance', 'Эндуранс'),
    ]

    name = models.CharField(max_length=200)
    description = models.TextField()
    race_type = models.CharField(max_length=20, choices=RACE_TYPE_CHOICES)
    track = models.CharField(max_length=100)
    start_date = models.DateTimeField()
    end_date = models.DateTimeField()
    max_participants = models.IntegerField()
    registration_deadline = models.DateTimeField()
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.name

    1.5 Модель регистрации

    class Registration(models.Model):
    STATUS_CHOICES = [
        ('pending', 'Ожидает'),
        ('approved', 'Одобрено'),
        ('rejected', 'Отклонено'),
        ('cancelled', 'Отменено'),
    ]

    driver = models.ForeignKey(Driver, on_delete=models.CASCADE)
    race = models.ForeignKey(Race, on_delete=models.CASCADE)
    registration_date = models.DateTimeField(auto_now_add=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='pending')
    notes = models.TextField(blank=True)

    class Meta:
        unique_together = ['driver', 'race']

    def __str__(self):
        return f"{self.driver} - {self.race}"

    1.6 Модель результатов

    class RaceResult(models.Model):
    registration = models.OneToOneField(Registration, on_delete=models.CASCADE)
    position = models.IntegerField()
    lap_time = models.DurationField()
    total_time = models.DurationField()
    points = models.IntegerField(default=0)
    is_finished = models.BooleanField(default=False)
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return f"{self.registration.driver} - Позиция {self.position}"

    1.7 Модель комментариев

    class Comment(models.Model):
    COMMENT_TYPE_CHOICES = [
        ('cooperation', 'Вопрос о сотрудничестве'),
        ('race', 'Вопрос о гонках'),
        ('other', 'Иное'),
    ]

    user = models.ForeignKey(CustomUser, on_delete=models.CASCADE)
    race = models.ForeignKey(Race, on_delete=models.CASCADE, related_name='comments')
    comment_type = models.CharField(max_length=20, choices=COMMENT_TYPE_CHOICES)
    text = models.TextField()
    rating = models.IntegerField(choices=[(i, i) for i in range(1, 11)])
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return f"{self.user.username} - {self.race.name}"

    2.1 Формы для регистрации

    from django import forms
from .models import CustomUser, Team, Driver, Race, Registration, Comment

class UserRegistrationForm(forms.ModelForm):
    password1 = forms.CharField(widget=forms.PasswordInput)
    password2 = forms.CharField(widget=forms.PasswordInput)

    class Meta:
        model = CustomUser
        fields = ['username', 'email', 'first_name', 'last_name', 'phone', 'birth_date', 'bio']
        widgets = {
            'birth_date': forms.DateInput(attrs={'type': 'date'}),
            'bio': forms.Textarea(attrs={'rows': 3}),
        }

    def clean_password2(self):
        password1 = self.cleaned_data.get('password1')
        password2 = self.cleaned_data.get('password2')
        if password1 and password2 and password1 != password2:
            raise forms.ValidationError('Пароли не совпадают')
        return password2

class TeamForm(forms.ModelForm):
    class Meta:
        model = Team
        fields = ['name', 'description', 'founded_year', 'country', 'website']
        widgets = {
            'description': forms.Textarea(attrs={'rows': 4}),
        }

class DriverForm(forms.ModelForm):
    class Meta:
        model = Driver
        fields = ['driver_number', 'experience', 'driver_class', 'car_description', 'bio']
        widgets = {
            'car_description': forms.Textarea(attrs={'rows': 3}),
            'bio': forms.Textarea(attrs={'rows': 4}),
        }

class RegistrationForm(forms.ModelForm):
    class Meta:
        model = Registration
        fields = ['notes']
        widgets = {
            'notes': forms.Textarea(attrs={'rows': 3}),
        }

class CommentForm(forms.ModelForm):
    class Meta:
        model = Comment
        fields = ['comment_type', 'text', 'rating']
        widgets = {
            'text': forms.Textarea(attrs={'rows': 4}),
        }

    from django.views.generic import ListView, DetailView, CreateView, UpdateView, DeleteView
from django.contrib.auth.mixins import LoginRequiredMixin
from django.contrib.auth import login
from django.shortcuts import render, redirect
from django.db.models import Q
from .models import Race, Driver, Team, Registration, Comment
from .forms import UserRegistrationForm, TeamForm, DriverForm, RegistrationForm, CommentForm

class HomeView(ListView):
    model = Race
    template_name = 'racing_app/home.html'
    context_object_name = 'races'
    paginate_by = 6

    def get_queryset(self):
        return Race.objects.filter(is_active=True).order_by('-start_date')

class RaceListView(ListView):
    model = Race
    template_name = 'racing_app/race_list.html'
    context_object_name = 'races'
    paginate_by = 12

    def get_queryset(self):
        queryset = Race.objects.filter(is_active=True)
        search = self.request.GET.get('search')
        if search:
            queryset = queryset.filter(
                Q(name__icontains=search) | 
                Q(track__icontains=search) |
                Q(description__icontains=search)
            )
        return queryset.order_by('-start_date')

class RaceDetailView(DetailView):
    model = Race
    template_name = 'racing_app/race_detail.html'
    context_object_name = 'race'

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        race = self.get_object()
        context['registrations'] = Registration.objects.filter(race=race)
        context['comments'] = Comment.objects.filter(race=race).order_by('-created_at')
        return context

class DriverListView(ListView):
    model = Driver
    template_name = 'racing_app/driver_list.html'
    context_object_name = 'drivers'
    paginate_by = 12

    def get_queryset(self):
        return Driver.objects.select_related('user', 'team').all()

class TeamListView(ListView):
    model = Team
    template_name = 'racing_app/team_list.html'
    context_object_name = 'teams'
    paginate_by = 12

    def get_queryset(self):
        return Team.objects.all().order_by('name')

    3.2 Представления для регистрации

    class RegistrationCreateView(LoginRequiredMixin, CreateView):
    model = Registration
    form_class = RegistrationForm
    template_name = 'racing_app/registration_form.html'

    def form_valid(self, form):
        race = Race.objects.get(pk=self.kwargs['race_id'])
        driver = Driver.objects.get(user=self.request.user)
        form.instance.driver = driver
        form.instance.race = race
        return super().form_valid(form)

    def get_success_url(self):
        return reverse('race_detail', kwargs={'pk': self.kwargs['race_id']})

class CommentCreateView(LoginRequiredMixin, CreateView):
    model = Comment
    form_class = CommentForm
    template_name = 'racing_app/comment_form.html'

    def form_valid(self, form):
        race = Race.objects.get(pk=self.kwargs['race_id'])
        form.instance.user = self.request.user
        form.instance.race = race
        return super().form_valid(form)

    def get_success_url(self):
        return reverse('race_detail', kwargs={'pk': self.kwargs['race_id']})

    Практическое задание 6 — Административная панель

    from django.contrib import admin
from .models import CustomUser, Team, Driver, Race, Registration, RaceResult, Comment

@admin.register(CustomUser)
class CustomUserAdmin(admin.ModelAdmin):
    list_display = ['username', 'email', 'first_name', 'last_name', 'is_organizer']
    list_filter = ['is_organizer', 'is_staff', 'is_active']
    search_fields = ['username', 'email', 'first_name', 'last_name']

@admin.register(Team)
class TeamAdmin(admin.ModelAdmin):
    list_display = ['name', 'country', 'founded_year']
    list_filter = ['country', 'founded_year']
    search_fields = ['name', 'country']

@admin.register(Driver)
class DriverAdmin(admin.ModelAdmin):
    list_display = ['user', 'team', 'driver_number', 'experience', 'driver_class']
    list_filter = ['experience', 'driver_class', 'team']
    search_fields = ['user__username', 'user__first_name', 'user__last_name']

@admin.register(Race)
class RaceAdmin(admin.ModelAdmin):
    list_display = ['name', 'race_type', 'track', 'start_date', 'is_active']
    list_filter = ['race_type', 'is_active', 'start_date']
    search_fields = ['name', 'track', 'description']

@admin.register(Registration)
class RegistrationAdmin(admin.ModelAdmin):
    list_display = ['driver', 'race', 'status', 'registration_date']
    list_filter = ['status', 'registration_date']
    search_fields = ['driver__user__username', 'race__name']

@admin.register(RaceResult)
class RaceResultAdmin(admin.ModelAdmin):
    list_display = ['registration', 'position', 'lap_time', 'points', 'is_finished']
    list_filter = ['is_finished', 'position']
    search_fields = ['registration__driver__user__username']

@admin.register(Comment)
class CommentAdmin(admin.ModelAdmin):
    list_display = ['user', 'race', 'comment_type', 'rating', 'created_at']
    list_filter = ['comment_type', 'rating', 'created_at']
    search_fields = ['user__username', 'race__name', 'text']
