    Практическая работа №2.2 — CRUD интерфейсы в Django

    Автор: Губанов Егор
    Группа: К3340

    Цель работы

    Изучить реализацию CRUD (Create, Read, Update, Delete) интерфейсов средствами Django Web Framework, освоить работу с формами, функциональными и классовыми представлениями, а также реализовать связи "многие-ко-многим" между моделями.

    Задача 1: Доработка модели данных. Реализация связи "Многие ко многим"

    1.1 Создание моделей

    Созданы модели для системы управления автомобилями и их владельцами:

    class Owner(models.Model):
    """Модель владельца автомобиля"""
    first_name = models.CharField(max_length=30, verbose_name='Имя')
    last_name = models.CharField(max_length=30, verbose_name='Фамилия')
    birth_date = models.DateField(verbose_name='Дата рождения')
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата создания')
    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')

    class Meta:
        verbose_name = 'Владелец'
        verbose_name_plural = 'Владельцы'
        ordering = ['last_name', 'first_name']

    def __str__(self):
        return f"{self.first_name} {self.last_name}"

class Car(models.Model):
    """Модель автомобиля"""
    make = models.CharField(max_length=20, verbose_name='Марка')
    model = models.CharField(max_length=20, verbose_name='Модель')
    color = models.CharField(max_length=30, verbose_name='Цвет')
    state_number = models.CharField(max_length=15, unique=True, verbose_name='Гос. номер')
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата создания')
    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')

    class Meta:
        verbose_name = 'Автомобиль'
        verbose_name_plural = 'Автомобили'
        ordering = ['make', 'model']

    def __str__(self):
        return f"{self.make} {self.model} ({self.state_number})"

class Ownership(models.Model):
    """Модель владения автомобилем (ассоциативная таблица)"""
    owner = models.ForeignKey(Owner, on_delete=models.CASCADE, verbose_name='Владелец')
    car = models.ForeignKey(Car, on_delete=models.CASCADE, verbose_name='Автомобиль')
    start_date = models.DateTimeField(verbose_name='Дата начала владения')
    end_date = models.DateTimeField(null=True, blank=True, verbose_name='Дата окончания владения')
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Дата создания')
    updated_at = models.DateTimeField(auto_now=True, verbose_name='Дата обновления')

    class Meta:
        verbose_name = 'Владение'
        verbose_name_plural = 'Владения'
        ordering = ['-start_date']
        unique_together = ['owner', 'car', 'start_date']

    def __str__(self):
        return f"{self.owner} - {self.car} ({self.start_date.date()})"

    @property
    def is_current(self):
        """Проверяет, является ли владение текущим"""
        return self.end_date is None

    1.2 Реализация связи "многие-ко-многим"

    Связь между владельцами и автомобилями реализована через промежуточную модель Ownership, которая содержит дополнительные поля: - start_date - дата начала владения - end_date - дата окончания владения (может быть пустой для текущего владения)

    Это позволяет отслеживать историю владения автомобилями и избегать пересечений по датам.

    Задача 2: Работа с представлениями

    2.1 Функциональные представления для владельцев

    Реализованы функциональные представления для всех CRUD операций с владельцами:

    def owner_list(request):
    """Функциональное представление для списка всех владельцев"""
    owners = Owner.objects.all()
    paginator = Paginator(owners, 10)  # 10 владельцев на страницу
    page_number = request.GET.get('page')
    page_obj = paginator.get_page(page_number)

    context = {
        'page_obj': page_obj,
        'owners': page_obj,
    }
    return render(request, 'crud_app/owner_list.html', context)

def owner_detail(request, pk):
    """Функциональное представление для детальной информации о владельце"""
    owner = get_object_or_404(Owner, pk=pk)
    ownerships = Ownership.objects.filter(owner=owner)

    context = {
        'owner': owner,
        'ownerships': ownerships,
    }
    return render(request, 'crud_app/owner_detail.html', context)

def owner_create(request):
    """Функциональное представление для создания нового владельца"""
    if request.method == 'POST':
        form = OwnerForm(request.POST)
        if form.is_valid():
            form.save()
            messages.success(request, 'Владелец успешно создан!')
            return redirect('owner_list')
    else:
        form = OwnerForm()

    context = {
        'form': form,
        'title': 'Создать владельца',
        'button_text': 'Создать',
    }
    return render(request, 'crud_app/owner_form.html', context)

    2.2 Классовые представления для автомобилей

    Реализованы классовые представления для всех CRUD операций с автомобилями:

    class CarListView(ListView):
    """Классовое представление для списка автомобилей"""
    model = Car
    template_name = 'crud_app/car_list.html'
    context_object_name = 'cars'
    paginate_by = 10

    def get_queryset(self):
        """Переопределение queryset для фильтрации"""
        queryset = Car.objects.all()
        search = self.request.GET.get('search')

        if search:
            queryset = queryset.filter(
                Q(make__icontains=search) |
                Q(model__icontains=search) |
                Q(state_number__icontains=search)
            )

        return queryset

class CarDetailView(DetailView):
    """Классовое представление для детальной информации об автомобиле"""
    model = Car
    template_name = 'crud_app/car_detail.html'
    context_object_name = 'car'

    def get_context_data(self, **kwargs):
        """Добавление дополнительного контекста"""
        context = super().get_context_data(**kwargs)
        car = self.get_object()
        context['ownerships'] = Ownership.objects.filter(car=car)
        return context

class CarCreateView(CreateView):
    """Классовое представление для создания автомобиля"""
    model = Car
    form_class = CarForm
    template_name = 'crud_app/car_form.html'
    success_url = reverse_lazy('car_list')

    def form_valid(self, form):
        """Добавление сообщения об успехе"""
        messages.success(self.request, 'Автомобиль успешно создан!')
        return super().form_valid(form)

    context = {}

    return render(request, "crud_app/list_view.html", context)

def create_view(request):
    """Пример функционального представления для создания"""
    context = {}
    form = OwnerForm(request.POST or None)
    if form.is_valid():
        form.save()
        messages.success(request, 'Владелец создан!')
        return redirect('owner_list')

    return render(request, "crud_app/create_view.html", context)

    3.1 Формы на основе функций

    Созданы формы для владельцев с использованием функциональных представлений:

    class OwnerForm(forms.ModelForm):
    """Форма для создания и редактирования владельца"""

    class Meta:
        model = Owner
        fields = ['first_name', 'last_name', 'birth_date']
        widgets = {
            'birth_date': forms.DateInput(attrs={'type': 'date'}),
        }
        labels = {
            'first_name': 'Имя',
            'last_name': 'Фамилия',
            'birth_date': 'Дата рождения',
        }

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        for field in self.fields.values():
            field.widget.attrs.update({'class': 'form-control'})
