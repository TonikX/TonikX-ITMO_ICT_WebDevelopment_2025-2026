    Задача 1: Расширение пользовательской модели

    1.1 Способы расширения пользовательской модели

    В Django существует 4 основных способа расширения пользовательской модели:

    1. Простое расширение модели (proxy) - без создания новых таблиц
    2. Связь один-к-одному с пользовательской моделью - через дополнительную модель
    3. Расширение AbstractBaseUser - полная замена модели пользователя
    4. Расширение AbstractUser - добавление полей к стандартной модели

    В данной работе используется расширение AbstractUser как наиболее практичный подход.

    1.2 Реализация кастомной модели пользователя

    Создана кастомная модель пользователя с дополнительными полями:

    class CustomUser(AbstractUser):
    """Кастомная модель пользователя с дополнительными полями"""
    passport_number = models.CharField(
        max_length=20, 
        unique=True, 
        verbose_name='Номер паспорта',
        help_text='Номер паспорта пользователя'
    )
    home_address = models.TextField(
        verbose_name='Домашний адрес',
        help_text='Полный домашний адрес пользователя'
    )
    nationality = models.CharField(
        max_length=100, 
        default='Российская', 
        verbose_name='Национальность',
        help_text='Национальность пользователя'
    )
    phone_number = models.CharField(
        max_length=20, 
        blank=True, 
        verbose_name='Номер телефона',
        help_text='Контактный номер телефона'
    )
    birth_date = models.DateField(
        null=True, 
        blank=True, 
        verbose_name='Дата рождения',
        help_text='Дата рождения пользователя'
    )
    avatar = models.ImageField(
        upload_to=get_upload_path, 
        blank=True, 
        null=True, 
        verbose_name='Аватар',
        help_text='Фото профиля пользователя'
    )
    is_car_owner = models.BooleanField(
        default=False, 
        verbose_name='Владелец автомобиля',
        help_text='Является ли пользователь владельцем автомобиля'
    )

    class Meta:
        verbose_name = 'Пользователь'
        verbose_name_plural = 'Пользователи'
        ordering = ['last_name', 'first_name']

    def __str__(self):
        return f"{self.first_name} {self.last_name} ({self.username})"

    @property
    def full_name(self):
        """Полное имя пользователя"""
        return f"{self.last_name} {self.first_name}"

    @property
    def age(self):
        """Возраст пользователя"""
        if self.birth_date:
            return timezone.now().year - self.birth_date.year
        return None

    def get_absolute_url(self):
        return reverse('user_detail', kwargs={'pk': self.pk})

    class CustomUserCreationForm(UserCreationForm):

        model = User

            'username', 'email', 'first_name', 'last_name', 
            'passport_number', 'home_address', 'nationality',
            'phone_number', 'birth_date', 'avatar', 'is_car_owner'
        )
        widgets = {
            'birth_date': forms.DateInput(attrs={'type': 'date'}),
            'home_address': forms.Textarea(attrs={'rows': 3}),
        }

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        for field in self.fields.values():
            field.widget.attrs.update({'class': 'form-control'})

        # Делаем поля обязательными
        self.fields['passport_number'].required = True
        self.fields['home_address'].required = True
        self.fields['nationality'].required = True
        self.fields['email'].required = True
        self.fields['first_name'].required = True
        self.fields['last_name'].required = True

    4.1 Классовые представления

    class UserListView(ListView):
    """Список пользователей"""
    model = CustomUser
    template_name = 'library_app/user_list.html'
    context_object_name = 'users'
    paginate_by = 20

    def get_queryset(self):
        """Фильтрация пользователей"""
        queryset = CustomUser.objects.filter(is_active=True)

        # Поиск
        search = self.request.GET.get('search')
        if search:
            queryset = queryset.filter(
                Q(first_name__icontains=search) |
                Q(last_name__icontains=search) |
                Q(email__icontains=search) |
                Q(passport_number__icontains=search) |
                Q(username__icontains=search)
            )

        # Фильтр по национальности
        nationality = self.request.GET.get('nationality')
        if nationality:
            queryset = queryset.filter(nationality=nationality)

        # Фильтр по владельцам автомобилей
        is_car_owner = self.request.GET.get('is_car_owner')
        if is_car_owner:
            queryset = queryset.filter(is_car_owner=True)

        return queryset.order_by('last_name', 'first_name')

class UserDetailView(DetailView):
    """Детальная информация о пользователе"""
    model = CustomUser
    template_name = 'library_app/user_detail.html'
    context_object_name = 'user'

    def get_context_data(self, **kwargs):
        context = super().get_context_data(**kwargs)
        user = self.get_object()

        # Владения пользователя
        context['ownerships'] = Ownership.objects.filter(owner=user).order_by('-start_date')

        # Статистика
        context['total_cars'] = context['ownerships'].count()
        context['current_cars'] = context['ownerships'].filter(status='current').count()

        return context

class UserCreateView(CreateView):
    """Создание пользователя"""
    model = CustomUser
    form_class = CustomUserCreationForm
    template_name = 'library_app/user_form.html'
    success_url = reverse_lazy('user_list')

    def form_valid(self, form):
        """Добавление сообщения об успехе"""
        messages.success(self.request, 'Пользователь успешно создан!')
        return super().form_valid(form)

    class CustomUserViewSet(viewsets.ModelViewSet):
    """ViewSet для пользователей"""
    queryset = CustomUser.objects.filter(is_active=True)
    serializer_class = CustomUserSerializer
    permission_classes = [IsAuthenticatedOrReadOnly]
    filter_backends = [DjangoFilterBackend, filters.SearchFilter, filters.OrderingFilter]
    search_fields = ['first_name', 'last_name', 'email', 'passport_number', 'username']
    ordering_fields = ['last_name', 'first_name', 'date_joined']
    ordering = ['last_name', 'first_name']

    def get_permissions(self):
        """Настройка прав доступа"""
        if self.action in ['create', 'update', 'partial_update', 'destroy']:
            permission_classes = [IsAdminUser]
        else:
            permission_classes = [IsAuthenticatedOrReadOnly]
        return [permission() for permission in permission_classes]

    @action(detail=True, methods=['get'])
    def cars(self, request, pk=None):
        """Получить автомобили пользователя"""
        user = self.get_object()
        ownerships = Ownership.objects.filter(owner=user)

        # Пагинация
        page = self.paginate_queryset(ownerships)
        if page is not None:
            serializer = OwnershipSerializer(page, many=True)
            return self.get_paginated_response(serializer.data)

        serializer = OwnershipSerializer(ownerships, many=True)
        return Response(serializer.data)
